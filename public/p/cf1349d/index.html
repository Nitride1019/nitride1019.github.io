#include <bits/stdc++.h>
#define ll long long
using namespace std;

const int M = 998244353;
ll qpow(ll a, ll b) {
	ll res = 1;
	while (b) {
		if (b & 1) res = res * a % M;
		a = a * a % M;
		b >>= 1;
	}
	return res;
}

const int maxn = 100005, maxm = 300005;
int n, m;
int a[maxn];
ll res[maxm];
int main() {
	cin.tie(0)->sync_with_stdio(0);
	cin >> n;
	for (int i = 1; i <= n; i++) {
		cin >> a[i];
		m += a[i];
	}

	ll invm = qpow(m, M - 2), invn1 = qpow(n - 1, M - 2);
	for (int i = m; i >= 1; i--) {
		ll k1 = i * invm % M * invn1 % M, k2 = (m - i) * invm % M;
		res[i] = (k2 * res[i + 1] + 1) % M * qpow(k1, M - 2) % M;
	}

	for (int i = 1; i <= m; i++) res[i] = (res[i] + res[i - 1]) % M;
	ll s = 0;
	for (int i = 1; i <= n; i++) s = (s + res[m - a[i]]) % M;
	s = (s + M - res[m] * (n - 1) % M) % M;
	cout << s * qpow(n, M - 2) % M;
}
